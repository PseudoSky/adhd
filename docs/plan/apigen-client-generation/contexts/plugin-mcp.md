## Goal

Implement the MCP plugin in `@adhd/apigen-plugin-mcp`. A single package covers all three transports: `stdio`, `sse`, and `streaming-http`. The plugin implements `generate()` (emit TypeScript server files) and `run()` (in-process server). After this state, `--type mcp --transport stdio` works end-to-end in both generate and run modes.

### `generate.ts`

```typescript
import type { PluginInput, PluginOutput } from '@adhd/apigen-core'
import { generateStdioServer } from './templates/server-stdio.tpl'
import { generateHttpServer } from './templates/server-http.tpl'
import { generateIndex } from './templates/index.tpl'

export function generate(input: PluginInput): PluginOutput {
  const transport = (input.options['transport'] as string) ?? 'stdio'
  const port = (input.options['port'] as number) ?? 3000
  const descriptions = (input.options['toolDescriptions'] as Record<string, string>) ?? {}

  const indexContent = generateIndex(input.packages, descriptions)
  const serverContent = transport === 'stdio'
    ? generateStdioServer()
    : generateHttpServer(transport, port)

  return {
    files: [
      { path: 'index.ts', content: indexContent },
      { path: 'server.ts', content: serverContent },
    ],
  }
}
```

### `templates/index.tpl.ts`

Generates `index.ts` — a file that exports `toolMetas`, `groupFns`, `groupCreateClient`. Template:

```typescript
export function generateIndex(
  packages: PluginInput['packages'],
  descriptions: Record<string, string>
): string {
  const lines: string[] = [
    `// Auto-generated by @adhd/apigen-cli — do not edit`,
  ]
  for (const pkg of packages) {
    lines.push(`import * as ${pkg.id}_fns from '${pkg.importPath}'`)
  }
  lines.push(``)
  lines.push(`export const toolMetas: Record<string, { group: string; schema: unknown }> = {`)
  for (const pkg of packages) {
    for (const [fnName, fnSchema] of Object.entries(pkg.schemas)) {
      const desc = descriptions[fnName] ?? fnName
      lines.push(`  ${fnName}: { group: '${pkg.id}', schema: ${JSON.stringify(fnSchema)} },`)
    }
  }
  lines.push(`}`)
  lines.push(``)
  lines.push(`export const groupFns: Record<string, Record<string, (...args: unknown[]) => unknown>> = {`)
  for (const pkg of packages) {
    lines.push(`  '${pkg.id}': ${pkg.id}_fns as Record<string, (...args: unknown[]) => unknown>,`)
  }
  lines.push(`}`)
  lines.push(``)
  lines.push(`export const groupCreateClient: Record<string, ((e: Record<string, unknown>) => Promise<unknown>) | undefined> = {`)
  for (const pkg of packages) {
    lines.push(`  '${pkg.id}': undefined, // injected at startup if createClient provided`)
  }
  lines.push(`}`)
  return lines.join('\n')
}
```

### `templates/server-stdio.tpl.ts`

Generates `server.ts` for stdio transport:

```typescript
export function generateStdioServer(): string {
  return `
import { Server } from '@modelcontextprotocol/sdk/server/index.js'
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js'
import { ListToolsRequestSchema, CallToolRequestSchema } from '@modelcontextprotocol/sdk/types.js'
import { dispatch } from '@adhd/apigen-runtime'
import { toolMetas, groupFns, groupCreateClient } from './index.js'

const server = new Server({ name: 'apigen-mcp', version: '1.0.0' }, { capabilities: { tools: {} } })

server.setRequestHandler(ListToolsRequestSchema, async () => ({
  tools: Object.entries(toolMetas).map(([name, meta]) => ({
    name,
    description: name,
    inputSchema: (meta.schema as any).input,
  })),
}))

server.setRequestHandler(CallToolRequestSchema, async (req) => {
  const { name, arguments: args = {} } = req.params
  const meta = toolMetas[name]
  if (!meta) throw new Error(\`Unknown tool: \${name}\`)
  const result = await dispatch(
    groupFns[meta.group],
    groupCreateClient[meta.group],
    meta.schema as any,
    name,
    args as Record<string, unknown>,
    ((args as any)['data'] ?? {}) as Record<string, unknown>,
  )
  return { content: [{ type: 'text', text: JSON.stringify(result) }] }
})

const transport = new StdioServerTransport()
server.connect(transport)
`
}
```

```typescript
export function generateHttpServer(transport: 'sse' | 'streaming-http', port: number): string {
  const TransportClass = transport === 'sse' ? 'SSEServerTransport' : 'StreamableHTTPServerTransport'
  const transportImport = transport === 'sse'
    ? `import { SSEServerTransport } from '@modelcontextprotocol/sdk/server/sse.js'`
    : `import { StreamableHTTPServerTransport } from '@modelcontextprotocol/sdk/server/streamableHttp.js'`

  return `
import { createServer } from 'node:http'
import { Server } from '@modelcontextprotocol/sdk/server/index.js'
${transportImport}
import { ListToolsRequestSchema, CallToolRequestSchema } from '@modelcontextprotocol/sdk/types.js'
import { dispatch } from '@adhd/apigen-runtime'
import { toolMetas, groupFns, groupCreateClient } from './index.js'

const mcpServer = new Server({ name: 'apigen-mcp', version: '1.0.0' }, { capabilities: { tools: {} } })

mcpServer.setRequestHandler(ListToolsRequestSchema, async () => ({
  tools: Object.entries(toolMetas).map(([name, meta]) => ({
    name, description: name, inputSchema: (meta.schema as any).input,
  })),
}))

mcpServer.setRequestHandler(CallToolRequestSchema, async (req) => {
  const { name, arguments: args = {} } = req.params
  const meta = toolMetas[name]
  const result = await dispatch(groupFns[meta.group], groupCreateClient[meta.group], meta.schema as any, name, args as any, ((args as any)['data'] ?? {}) as any)
  return { content: [{ type: 'text', text: JSON.stringify(result) }] }
})

const transport = new ${TransportClass}({ endpoint: '/mcp' } as any)
const httpServer = createServer((req, res) => transport.handleRequest(req, res))
httpServer.listen(${port}, () => console.error('MCP ${transport} server on port ${port}'))
`
}
```

### `run.ts`

In-process server for all transports. Uses `node:http` for HTTP transports (no fastify/express dep):

```typescript
import { Server } from '@modelcontextprotocol/sdk/server/index.js'
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js'
import { SSEServerTransport } from '@modelcontextprotocol/sdk/server/sse.js'
import { StreamableHTTPServerTransport } from '@modelcontextprotocol/sdk/server/streamableHttp.js'
import { ListToolsRequestSchema, CallToolRequestSchema } from '@modelcontextprotocol/sdk/types.js'
import { dispatch } from '@adhd/apigen-runtime'
import type { RunInput } from '@adhd/apigen-core'
import { createServer } from 'node:http'

export async function run(input: RunInput): Promise<void> {
  const transport = (input.options['transport'] as string) ?? 'stdio'
  const port = (input.options['port'] as number) ?? 3000
  const descriptions = (input.options['toolDescriptions'] as Record<string, string>) ?? {}

  const server = new Server({ name: 'apigen-mcp', version: '1.0.0' }, { capabilities: { tools: {} } })

  // Build toolMetas from all packages
  const toolMetas: Record<string, { group: string; schema: unknown }> = {}
  for (const pkg of input.packages) {
    for (const [fnName, schema] of Object.entries(pkg.schemas)) {
      toolMetas[fnName] = { group: pkg.id, schema }
    }
  }

  server.setRequestHandler(ListToolsRequestSchema, async () => ({
    tools: Object.entries(toolMetas).map(([name, meta]) => ({
      name,
      description: descriptions[name] ?? name,
      inputSchema: (meta.schema as any).input,
    })),
  }))

  server.setRequestHandler(CallToolRequestSchema, async (req) => {
    const { name, arguments: args = {} } = req.params
    const meta = toolMetas[name]
    if (!meta) throw new Error(`Unknown tool: ${name}`)
    const pkg = input.packages.find(p => p.id === meta.group)!
    const result = await dispatch(
      pkg.fns!,
      pkg.createClient,
      meta.schema as any,
      name,
      args as Record<string, unknown>,
      ((args as any)['data'] ?? {}) as Record<string, unknown>,
    )
    return { content: [{ type: 'text', text: JSON.stringify(result) }] }
  })

  if (transport === 'stdio') {
    const t = new StdioServerTransport()
    await server.connect(t)
    return new Promise((resolve) => {
      if (input.signal) input.signal.addEventListener('abort', resolve)
    })
  } else {
    const TransportClass = transport === 'sse' ? SSEServerTransport : StreamableHTTPServerTransport
    const mcpTransport = new TransportClass({ endpoint: '/mcp' } as any)
    const httpServer = createServer((req, res) => (mcpTransport as any).handleRequest(req, res))
    httpServer.listen(port)
    return new Promise((resolve) => {
      if (input.signal) input.signal.addEventListener('abort', () => httpServer.close(resolve))
    })
  }
}
```
